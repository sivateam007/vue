Vue Component Instance

Vue.js இல் கூறு நிகழ்வு பொருள்கள் மற்றும் முறைகளைக் கற்றுக்கொள்ளுங்கள்

பொருள்கள்

Vue க்கு கூறு நிகழ்வில் உள்ள பொருள்களின் தொகுப்பு உள்ளது.

பொருள் விளக்கம்
$attrs கூறு டேக்கில் அமைக்கப்பட்ட பாழ்விழும் பண்புக்கூறுகள் மற்றும் நிகழ்வு கேட்பான்களைக் குறிக்கிறது.
$data Vue நிகழ்வின் தரவுப் பகுதியில் சேமிக்கப்பட்ட பண்புகளைக் குறிக்கிறது.
$el Vue கூறின் ரூட் DOM முனையைக் குறிக்கிறது.
$parent பெற்றோர் கூறின் Vue நிகழ்வைக் குறிக்கிறது.
$props பெறும் கூறில் அறிவிக்கப்பட்ட புரொப்ஸ் களைக் குறிக்கிறது.
$refs உள்ளமைக்கப்பட்ட 'ref' பண்புக்கூறுடன் குறிக்கப்பட்ட DOM உறுப்புகளைக் குறிக்கிறது.
$root மொத்த Vue பயன்பாட்டின் ரூட் கூறின் Vue நிகழ்வைக் குறிக்கிறது.
$slots பெற்றோர் கூறால் வழங்கப்பட்ட slots களைக் குறிக்கிறது.

பொருள்கள் விரிவான விளக்கம்

$attrs

பாழ்விழும் பண்புக்கூறுகள் மற்றும் நிகழ்வு கேட்பான்கள்

this.$attrs.class
this.$attrs.style
this.$attrs.onClick

$data

கூறு தரவுப் பண்புகள்

data() {
  return {
    message: 'Hello'
  }
}

// அணுகல்:
this.message
// அல்லது
this.$data.message

$el

ரூட் DOM முனை

// கூறின் ரூட் உறுப்பு:
this.$el

// DOM செயல்பாடுகள்:
this.$el.querySelector()
this.$el.style.backgroundColor

$parent

பெற்றோர் கூறு நிகழ்வு

// பெற்றோர் தரவுக்கு அணுகல்:
this.$parent.parentData

// பெற்றோர் முறைகளை அழைத்தல்:
this.$parent.parentMethod()

$props

பெறப்பட்ட புரொப்ஸ் கள்

props: ['title', 'count']

// அணுகல்:
this.title
// அல்லது
this.$props.title
this.$props.count

$refs

DOM உறுப்பு குறிப்புகள்

<input ref="myInput">

// அணுகல்:
this.$refs.myInput
this.$refs.myInput.focus()

$root

ரூட் கூறு நிகழ்வு

// முழு பயன்பாட்டின் ரூட்:
this.$root

// குளோபல் நிலை மேலாண்மை:
this.$root.globalState
this.$root.globalMethod()

$slots

பெற்றோர் வழங்கிய slots

// slots இருப்பதைச் சரிபார்க்க:
if (this.$slots.default) {
  // இயல்புநிலை slot உள்ளது
}

if (this.$slots.header) {
  // header slot உள்ளது
}

முறைகள்

Vue க்கு கூறு நிகழ்வில் உள்ள முறைகளின் தொகுப்பு உள்ளது, இது 'this' வார்த்தையுடன் அடையலாம்.

முறை விளக்கம்
$emit() பெற்றோர் கூறுக்குத் தகவல் தெரிவிக்கப் பயன்படும் தனிபயன் நிகழ்வைத் தொடங்குகிறது
$forceUpdate() Vue பயன்பாட்டின் மறு-ரெண்டரிங்கைக் கட்டாயப்படுத்துகிறது
$nextTick() தற்போதைய Vue கூறின் DOM புதுப்பிப்பு சுழற்சி முடியும் வரை காத்திருக்கிறது
$watch() கண்காணிப்பாளர்களை உருவாக்கப் பயன்படுகிறது, மேலும் கண்காணிப்பாளரை நிறுத்த நாம் பயன்படுத்தக்கூடிய நிறுத்தும் செயல்பாட்டை வழங்குகிறது

முறைகள் விரிவான விளக்கம்

$emit()

பெற்றோருக்குத் தகவல் அனுப்புதல்

// குழந்தை கூறில்:
methods: {
  sendData() {
    this.$emit('custom-event', data)
  }
}

// பெற்றோர் கூறில்:
<child-component 
  @custom-event="handleEvent"
/>

$forceUpdate()

கட்டாய மறு-ரெண்டரிங்

// கூறை கட்டாயமாக மீண்டும் ரெண்டர் செய்ய:
this.$forceUpdate()

// பயன்பாடு:
methods: {
  forceRerender() {
    // Reactivity அமைப்புகளைத் தவிர்க்க
    this.$forceUpdate()
  }
}

⚠️ எச்சரிக்கை:

கடைசி முயற்சியாக மட்டுமே $forceUpdate() பயன்படுத்தவும். பெரும்பாலும் நல்ல வடிவமைப்பு இதைத் தவிர்க்கும்.

$nextTick()

DOM புதுப்பிப்புக்குக் காத்திருத்தல்

// DOM புதுப்பிப்புக்குப் பிறகு:
this.someData = 'new value'

this.$nextTick(() => {
  // DOM இப்போது புதுப்பிக்கப்பட்டது
  this.$refs.myElement.focus()
})

// Promise தொடரியல்:
await this.$nextTick()
// DOM புதுப்பிக்கப்பட்டது

$watch()

தரவு மாற்றங்களைக் கண்காணித்தல்

// கண்காணிப்பாளரை உருவாக்க:
const unwatch = this.$watch(
  'counter',
  (newVal, oldVal) => {
    console.log(`Counter changed: ${oldVal} → ${newVal}`)
  }
)

// கண்காணிப்பாளரை நிறுத்த:
unwatch()

// உடனடி கண்காணிப்பாளர்:
this.$watch(
  'user.name',
  (newName) => {
    console.log(`Name changed to: ${newName}`)
  },
  { immediate: true }
)

நடைமுறை எடுத்துக்காட்டுகள்

1. $emit() உடன் பெற்றோர்-குழந்தை தொடர்பு

// ChildComponent.vue
<template>
  <button @click="notifyParent">
    Click Me
  </button>
</template>

<script>
export default {
  methods: {
    notifyParent() {
      this.$emit('button-clicked', {
        timestamp: new Date(),
        message: 'Button was clicked'
      })
    }
  }
}
</script>

// ParentComponent.vue
<template>
  <child-component 
    @button-clicked="handleButtonClick"
  />
</template>

<script>
export default {
  methods: {
    handleButtonClick(eventData) {
      console.log('Child button clicked:', eventData)
    }
  }
}
</script>

2. $refs உடன் DOM உறுப்புகளுக்கு அணுகல்

<template>
  <div>
    <input ref="emailInput" v-model="email">
    <button @click="focusInput">Focus Input</button>
  </div>
</template>

<script>
export default {
  data() {
    return {
      email: ''
    }
  },
  methods: {
    focusInput() {
      // $refs மூலம் DOM உறுப்புக்கு அணுகல்
      this.$refs.emailInput.focus()
      this.$refs.emailInput.select()
    }
  },
  mounted() {
    // கூறு mount செய்யப்படும் போது தானாக focus செய்ய
    this.$nextTick(() => {
      this.$refs.emailInput.focus()
    })
  }
}
</script>

Vue பயிற்சிகள்

பயிற்சிகளுடன் உங்களை சோதிக்கவும்

❓ குழந்தை கூறிலிருந்து பெற்றோர் கூறுக்குத் தகவல் அனுப்ப எந்த கூறு நிகழ்வு முறை பயன்படுத்தப்படுகிறது?

$send()
✗ தவறு! $send() என்பது Vue இல் கூறு நிகழ்வு முறை அல்ல
$parent()
✗ தவறு! $parent என்பது ஒரு பொருள், முறை அல்ல, மேலும் இது பெற்றோருக்குத் தகவல் அனுப்பப் பயன்படுவதில்லை
$emit()
✓ சரி! $emit() முறை குழந்தை கூறிலிருந்து பெற்றோர் கூறுக்குத் தகவல் அனுப்ப தனிபயன் நிகழ்வுகளைத் தொடங்க பயன்படுகிறது
$props()
✗ தவறு! $props என்பது ஒரு பொருள், முறை அல்ல, மேலும் இது பெற்றோரிடமிருந்து தகவலைப் பெற பயன்படுகிறது, அனுப்ப அல்ல
